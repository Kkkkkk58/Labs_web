Это мой скрипт для первой лабы по администрированию Windows Server.


@echo off

:begin
set /p input="Enter a string containing 4 symbols without whitespaces: "
if "%input%"=="q" (
    goto :eof
)
CALL :validate_input
if %ERRORLEVEL% neq 0 (
    goto retry
)

CALL :get_user
if %ERRORLEVEL% neq 0 (
    goto retry
)
CALL :get_group
if %ERRORLEVEL% neq 0 (
    goto retry
)

net user "%user%" /add
net localgroup "%group%" /add
net localgroup "%group%" "%user%" /add
net user "%user%" /active:yes

goto :eof


:retry
echo Try again!
goto begin

:validate_input
CALL :strlen %input%
if %strlength% neq 4 (
    echo Given string is not 4 characters long!
    exit /b 1
)
if not "%input%"=="%input: =%" (
    echo Given string contains whitespaces!
    exit /b 1
)
exit /b 0

:get_user
set "user=UPart2%input%"
net user "%user%" >nul 2>&1
if %ERRORLEVEL% equ 0 (
    echo User %user% already exists!
    exit /b 1
)
exit /b 0

:get_group
set "group=GPart2%input%"
net localgroup "%group%" >nul 2>&1
if %ERRORLEVEL% equ 0 (
    echo Group %group% already exists!
    exit /b 1
)
exit /b 0

:strlen
echo %1 > tempfile.txt
for %%? in (tempfile.txt) do ( set /a strlength=%%~z? - 3 )
del tempfile.txt
exit /b
